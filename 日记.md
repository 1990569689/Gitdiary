```python
def markdown_parser(text: str) -> str:
    """
    简易Markdown解析器示例
    """
    import re
    
    # 标题转换
    text = re.sub(r'^### (.*)$', r'<h3>\1</h3>', text, flags=re.MULTILINE)
    
    return f"""
    <!DOCTYPE html>
    <html>
    <body>
    {text}
    </body>
    </html>
    """
    
if __name__ == "__main__":
    sample = "### 这是一个测试"
    print(markdown_parser(sample))
```
